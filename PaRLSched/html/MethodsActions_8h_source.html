29  /*
30  * We assume here that there are two kinds of resources (i.e., main resources and child resources)

48  unsigned int action_per_main_source_; // e.g., this is per node running average performance
49  unsigned int previous_action_per_main_source_;

175  std::vector<unsigned int> compute_initial_action
176  (
177  const std::vector< std::vector < unsigned int> >& vec_num_child_sources
178  , const std::vector< unsigned int >& vec_max_num_child_sources_user
179  , const unsigned int& thread_num
180  , const unsigned int& max_number_available_main_sources
181  )
182  {
183  // We first compute the correct placement of this thread into the set of CPU's available
184  unsigned int overall_counter_cpu = 0;
185  unsigned int counter_cpu = 0;
186  unsigned int counter_numa = 0;
187  bool found_cpu = false;
188 
189  std::cout << " number of NUMA nodes used in initialization = " << vec_num_child_sources.size() << std::endl;
190 
191  while (!found_cpu)
192  {
193  counter_numa = 0;
194 
195  for (unsigned int r = 0; r < max_number_available_main_sources; r++)
196  {
197  // this is a loop over the main resource (e.g., NUMA nodes)
198  counter_cpu = 0;
199  unsigned int max_num_child_sources_available = std::min<unsigned int>(vec_num_child_sources[r].size(), vec_max_num_child_sources_user[r]);
200  // unsigned int max_num_child_sources = vec_num_child_sources[r].size();
201 
202  for (unsigned int j=0; j < max_num_child_sources_available; j++)
203  {
204  if ( thread_num == overall_counter_cpu)
205  {
206  // if the thread number coincides with the cpu number, then
207  // we will assign this cpu to the thread
208  found_cpu = true;
209  break;
210  }
211  overall_counter_cpu ++;
212  counter_cpu++;
213  }
214 
215  if (found_cpu)
216  break;
217  else{
218  counter_numa ++;
219  }
220 
221  }
222  }
223 
224  return {counter_numa, counter_cpu};

234  std::vector<unsigned int> compute_initial_action_rr
235  (
236  const std::vector< std::vector < unsigned int> >& vec_num_child_sources

238  , const unsigned int& thread_num
239  )
240  {
241  // We first compute the correct placement of this thread into the set of CPU's available
242  unsigned int numa_nodes = vec_num_child_sources.size();
243  std::cout << " number of NUMA nodes used in initialization = " << numa_nodes << std::endl;
244  std::vector<unsigned int> basis;
245  for (unsigned int r=0;r<numa_nodes;r++)
246  basis.push_back(r+1);
247 
248  unsigned int minimum_outcome(1e+10);
249  unsigned int arg_minimum(0);
250  for (unsigned int r=0;r<numa_nodes;r++)
251  {
252  // We compute the division result:
253  double outcome = (thread_num+1) / basis[r];
254  unsigned int remainder = (thread_num+1) % basis[r];
255  if (remainder == 0 && outcome < minimum_outcome)
256  {
257  minimum_outcome = outcome;
258  arg_minimum = r;
259  }
260  }
261 
262  // After selecting the best numa node, we select the best cpu node
263  unsigned int max_num_child_sources_available = std::min<unsigned int>(vec_num_child_sources[arg_minimum].size(), vec_max_num_child_sources_user[arg_minimum]);
264  unsigned int minimum_outcome2(1e+10);
265  unsigned int arg_minimum2(0);
266  unsigned int counter_cpu(0);
267  for (unsigned int j=0; j < max_num_child_sources_available; j++)
268  {
269  counter_cpu++;
270  double outcome2 = minimum_outcome / counter_cpu;
271  unsigned int remainder2 = minimum_outcome % counter_cpu;
272  if (remainder2 == 0 && outcome2 < minimum_outcome2)
273  {
274  minimum_outcome2 = outcome2;
275  arg_minimum2 = vec_num_child_sources[arg_minimum][j];
276  }
277  }
278 
279  std::cout << " allocation of thread " << thread_num << " is ( " << arg_minimum << " , " << arg_minimum2 << " ) " << std::endl;
280 
281  return {arg_minimum, arg_minimum2};
282 
283  }

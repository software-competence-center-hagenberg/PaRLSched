43 /*
44  * Struct_Estimate
45  *
46  * @description: provides the necessary information for keeping track our estimates for the performance of a thread over a specific node/cpu
47  * or the performance of a thread's memory over a specific node (or some further refinement if possible)
48  *
49  * This structure is planned for 'per-thread' use, since it will hold estimates over the proper use of resources by each thread separately.
50  *
51  * Within this structure, there is an additional struct that tracks the performances of the threads throughout the running time.
52  *
53  */
54 struct Struct_Estimate
55 {
56 
57  std::string resource_; /* The name of the resource assigned. */
58  unsigned int num_sources_; /* This is the number of the sources available, e.g., number of CPU's available. */
59  std::vector< std::vector< unsigned int > > vec_child_sources_; /* This is the number of the child sources available for each one of the main sources.
60  Potentially, there might be different number of child sources for each main source. */
61  std::vector<double> vec_estimates_; /* This vector holds our estimates over the best assignment, e.g., over the CPU assignment */
62  std::vector<double> vec_cummulative_estimates_;
63 
64  double run_average_performance_; /* It holds our running average performance */
65 
66  std::vector< Struct_Estimate > vec_child_estimates_; /* child estimates (used under hierarchical resources) */
67 
68  double low_benchmark_;
69  double high_benchmark_;
70 
71  double maximum_performance_;
72  bool random_switch_; /* This variable enforces random switch of an action */
73  bool action_change_;

75  /*
76  * This function initializes the structure
77  */
78  void initialize(
79  const std::string& resource
80  , const unsigned int& num_sources
81  , const unsigned int& max_num_sources_user
82  , const unsigned int& initial_action, const bool& mixed
83  )
84  {
85  random_switch_ = false;
86  action_change_ = false;
87  maximum_performance_ = 0;
88  resource_ = resource;
89  num_sources_ = std::min<unsigned int>(num_sources,max_num_sources_user);
90  vec_estimates_.resize(num_sources_);
91  vec_cummulative_estimates_.resize(num_sources_);
92  if (num_sources > 0)
93  {
94  for (unsigned int e=0;e<num_sources_;e++)
95  {
96  if (!mixed){
97  if (e == initial_action)
98  vec_estimates_[e] = 0.98;
99  else{
100  if (num_sources_ - 1 > 0)
101  vec_estimates_[e] = 0.02 / ((double)num_sources_ - 1);
102  else
103  vec_estimates_[e] = 0.02;
104  }
105  }
106  else
107  vec_estimates_.at(e) = (double)1/((double)num_sources);
108 
109  if (e==0)
110  vec_cummulative_estimates_[e] = vec_estimates_[e];
111  else
112  vec_cummulative_estimates_[e] = vec_cummulative_estimates_[e-1] + vec_estimates_[e];
113  }
114  }
115  }

118  /*
119  * initialize_w_child
120  * @description: It is implicitly assumed that the main resource also has a child resource
121  * For example, in the case of PROCESSING_BANDWIDTH, the main sources correspond to the NUMA nodes (if available)
122  * while the child sources correspond to the CPU units available in each NUMA node.

124  void initialize_w_child

126  const std::string& resource

128  , const unsigned int& max_num_sources_user
129  , const std::string& child_resource
130  , const std::vector< std::vector< unsigned int > >& vec_num_child_sources
131  , const std::vector< unsigned int >& vec_max_num_child_sources_user
132  , const unsigned int & thread_num
133  )
134  {
135  resource_ = resource;
136  num_sources_ = std::min<unsigned int>(num_sources,max_num_sources_user);
137  vec_estimates_.resize(num_sources_);
138  vec_cummulative_estimates_.resize(num_sources_);
139  vec_child_sources_.clear();
140  for (unsigned int r=0;r < num_sources_; r++){
141  std::vector<unsigned int> temp_child_resources;
142  unsigned int num_child_res = std::min<unsigned int>(vec_num_child_sources[r].size(),vec_max_num_child_sources_user[r]);
143  for (unsigned int cr=0; cr < num_child_res; cr++)
144  temp_child_resources.push_back(vec_num_child_sources[r][cr]);
145  vec_child_sources_.push_back(temp_child_resources);
146  }
147 
148  /* Computing the proper initial action based on Round-Robin */
149  Struct_Actions actions;
150  std::vector<unsigned int> initial_action = actions.compute_initial_action(vec_num_child_sources,vec_max_num_child_sources_user,thread_num, num_sources_);
151  unsigned int initial_action_main = initial_action[0];
152  unsigned int initial_action_child = initial_action[1];
153 
154  /* initialize the vector of estimates for the main resource */
155  if (num_sources > 0){
156  for (unsigned int e=0;e<num_sources_;e++)
157  {
158  if (num_sources_ - 1 > 0){
159  if (e == initial_action_main)
160  vec_estimates_[e] = 0.9999;
161  else{
162  vec_estimates_[e] = 0.0001 / ((double)num_sources_ - 1);
163  }
164  }
165  else
166  vec_estimates_[e] = 1;
167 
168  // vec_estimates_.at(e) = (double)1/((double)num_sources_);
169  if (e==0)
170  vec_cummulative_estimates_[e] = vec_estimates_[e];
171  else
172  vec_cummulative_estimates_[e] = vec_cummulative_estimates_[e-1] + vec_estimates_[e];
173  }
174  }
175  /* initialize the vector of estimates for the child resources */
176  vec_child_estimates_.clear();
177  if (vec_child_sources_.size() > 0){
178  // in case the number of the child resources is nonzero
179  for (unsigned int cr = 0; cr < vec_child_sources_.size(); cr++)
180  {
181  Struct_Estimate s_estimate;
182  if (cr == initial_action_main)
183  s_estimate.initialize(child_resource, vec_child_sources_[cr].size(), vec_max_num_child_sources_user[cr], initial_action_child, false);
184  else
185  // in case this is not the main source selected, then we mix the estimates over the child resources.
186  s_estimate.initialize(child_resource, vec_child_sources_[cr].size(), vec_max_num_child_sources_user[cr], initial_action_child, true);
187  vec_child_estimates_.push_back(s_estimate);
188  }
189  }
190  }

128  void initialize_w_child
129  (
130  const std::string& resource
131  , const unsigned int& num_sources
132  , const unsigned int& max_num_sources_user
133  , const std::string& child_resource
134  , const std::vector< std::vector< unsigned int > >& vec_num_child_sources
135  , const std::vector< unsigned int >& vec_max_num_child_sources_user
136  , const unsigned int & thread_num
137  )
138  {
139  resource_ = resource;
140  num_sources_ = std::min<unsigned int>(num_sources,max_num_sources_user);
141  vec_estimates_.resize(num_sources_);
142  vec_cummulative_estimates_.resize(num_sources_);
143  vec_child_sources_.clear();
144  for (unsigned int r=0;r < num_sources_; r++){
145  std::vector<unsigned int> temp_child_resources;
146  unsigned int num_child_res = std::min<unsigned int>(vec_num_child_sources[r].size(),vec_max_num_child_sources_user[r]);
147  for (unsigned int cr=0; cr < num_child_res; cr++)
148  temp_child_resources.push_back(vec_num_child_sources[r][cr]);
149  vec_child_sources_.push_back(temp_child_resources);
150  }
151 
152  /* Computing the proper initial action based on Round-Robin */
153  Struct_Actions actions;
154  std::vector<unsigned int> initial_action = actions.compute_initial_action(vec_num_child_sources,vec_max_num_child_sources_user,thread_num, num_sources_);
155  unsigned int initial_action_main = initial_action[0];
156  unsigned int initial_action_child = initial_action[1];
157 
158  /* initialize the vector of estimates for the main resource */
159  if (num_sources > 0){
160  for (unsigned int e=0;e<num_sources_;e++)
161  {
162  if (num_sources_ - 1 > 0){
163  if (e == initial_action_main)
164  vec_estimates_[e] = 0.9999;
165  else{
166  vec_estimates_[e] = 0.0001 / ((double)num_sources_ - 1);
167  }
168  }
169  else
170  vec_estimates_[e] = 1;
171 
172  if (e==0)
173  vec_cummulative_estimates_[e] = vec_estimates_[e];
174  else
175  vec_cummulative_estimates_[e] = vec_cummulative_estimates_[e-1] + vec_estimates_[e];
176  }
177  }
178  /* initialize the vector of estimates for the child resources */
179  vec_child_estimates_.clear();
180  if (vec_child_sources_.size() > 0){
181  // in case the number of the child resources is nonzero
182  for (unsigned int cr = 0; cr < vec_child_sources_.size(); cr++)
183  {
184  Struct_Estimate s_estimate;
185  if (cr == initial_action_main)
186  s_estimate.initialize(child_resource, vec_child_sources_[cr].size(), vec_max_num_child_sources_user[cr], initial_action_child, false);
187  else
188  // in case this is not the main source selected, then we mix the estimates over the child resources.
189  s_estimate.initialize(child_resource, vec_child_sources_[cr].size(), vec_max_num_child_sources_user[cr], initial_action_child, true);
190  vec_child_estimates_.push_back(s_estimate);
191  }
192  }
193  }

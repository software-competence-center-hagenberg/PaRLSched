250  /*
251  * We assume here that there are two kinds of resources (i.e., main resources and child resources)
252  * For example, a NUMA node may refer to as the main resource, and the underlying CPU may refer to as a child resource.
253  * For now, it is assumed that there is a 'single' child resource for each main resource, however the current setup
254  * may as well be expanded to the case of multiple child resources.
255  */
256  std::string resource_; // This is the type of resource over which these measurements are recorded.
257  unsigned int num_sources_; // This is the number of choices for the main resource
258 
259  /*
260  * Current Performances
261  */
262  double performance_; // Current performance of the thread w.r.t. the resource at hand.
263  double balanced_performance_; // A special type of performance that takes into account the variance from the mean (over all other threads).
264  double alg_performance_; // A special type of performance (to be defined).
265 
266  bool performance_update_ind_; // Indicates whether the performance has been updated
267 
268  /*
269  * Overall performance indicators (over all threads)
270  */
271  double overall_performance_;
272  double overall_balanced_performance_;
273 
274 
275  /*
276  * Statistical Properties
277  */
278  double min_performance_; // minimum performance observed so far
279  double max_performance_; // maximum performance observed so far
280 
281  /*
282  * Running Average Performances Per Main Resource
283  */
284  double run_average_performance_; // running average performance over the main resource
285  double run_average_balanced_performance_; // running average balanced performance over the main resource
286 
287  double run_average_balanced_performance_before_;
288 
289 
290  /*
291  * Running Average Performances Per Main & Child Source
292  *
293  * The idea here is that we would like to keep more specialized information about the average performances of each one of the Sources of
294  * either the main Resource or the child resources.
295  *
296  */
297  std::vector<unsigned int> vec_num_child_sources_;
298  std::vector< double > vec_run_average_performances_per_main_resource_; // e.g., this is per node running average performance
